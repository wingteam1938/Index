</div>
`;

function showNumber(){

document.getElementById("content").innerHTML=`
<div class="card">
<h3>Number Search</h3>

<input id="number"
placeholder="9876543210">

<button class="btn"
onclick="searchNumber()">
Search
</button>
</div>
`;

}

async function searchVehicle(){

let vehicle =
document.getElementById("vehicle").value;

let api =
"https://YOUR_API_URL?vehicle="+vehicle;

let res = await fetch(api);

let data = await res.json();

document.getElementById("content").innerHTML=`
<div class="card">
<h3>Vehicle Details</h3>

<p><b>Owner:</b>
${data.owner}</p>

<p><b>Vehicle:</b>
${data.vehicle}</p>

<p><b>State:</b>
${data.state}</p>

<p><b>RC Status:</b>
${data.status}</p>
</div>
`;

}

async function searchNumber(){

let number =
document.getElementById("number").value;

let api =
"https://YOUR_API_URL?number="+number;

let res = await fetch(api);

let data = await res.json();

document.getElementById("content").innerHTML=`
<div class="card">
<h3>Number Details</h3>

<pre>
${JSON.stringify(data,null,2)}
</pre>
</div>
`;

}
